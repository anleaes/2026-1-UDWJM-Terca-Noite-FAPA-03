@startuml UrbanizationApp

abstract class Person {
 -first_name: Char
 -last_name: Char
 -national_id: Char
 -address: Char
 -phone: Char
 -email: Email
 +toString()
}

class Employee {
 -salary: Decimal
 -position: Char
 -hire_date: Date
 -is_active: Boolean
 -company: ForeignKey[Company]
 +toString()
}

class Citizen {
 -registration_number: Char
 -neighborhood_association: Char
 +toString()
}

class Company {
 -corporate_name: Char
 -corporate_tax_id: Char
 -email: Email
 -phone: Char
 -is_active: Boolean
 +toString()
}

class Location {
 -name: Char
 -neighborhood: Char
 -city: Char
 -type: CharChoices
 -is_paved: Boolean
 +toString()
}

class Construction {
 -title: Char
 -type: CharChoices
 -status: CharChoices
 -start_date: Date
 -expected_end_date: Date
 -total_value: Decimal
 -is_completed: Boolean
 -photo: Image
 -report: File
 -location: ForeignKey[Location]
 -company: ForeignKey[Company]
 +toString()
}

class ConstructionEmployee {
 -construction: ForeignKey[Construction]
 -employee: ForeignKey[Employee]
}

class ConstructionEquipment {
 -construction: ForeignKey[Construction]
 -equipment: ForeignKey[Equipment]
 -allocation_date: Date
 -return_date: Date
 -usage_cost: Decimal
 +toString()
}

class Contract {
 -number: Char
 -signing_date: Date
 -value: Decimal
 -validity_days: Integer
 -document: File
 -is_active: Boolean
 -construction: ForeignKey[Construction]
 -company: ForeignKey[Company]
 +toString()
}

class Inspection {
 -visit_date: DateTime
 -status_found: CharChoices
 -description: Text
 -is_compliant: Boolean
 -score: Float
 -construction: ForeignKey[Construction]
 -employee: ForeignKey[Employee]
 +toString()
}

class AuditReport {
 -number: Char
 -issue_date: Date
 -conclusion: Text
 -is_approved: Boolean
 -inspection: OneToOne[Inspection]
 +toString()
}

class Incident {
 -description: Text
 -type: CharChoices
 -occurred_at: DateTime
 -severity: Integer
 -is_resolved: Boolean
 -construction: ForeignKey[Construction]
 -reported_by: ForeignKey[Citizen]
 +toString()
}

class Equipment {
 -name: Char
 -type: CharChoices
 -daily_rate: Decimal
 -is_available: Boolean
 -company: ForeignKey[Company]
 +toString()
}

' Relacionamentos Estruturais
Location *--> Construction : 1..n
Company *--> Construction : 1..n
Company *--> Contract : 1..n
Company *--> Employee : 1..n
Company *--> Equipment : 1..n

' Relacionamentos da Obra
Construction *--> Contract : 1..n
Construction *--> Inspection : 0..n
Construction *--> Incident : 0..n

' Classes Associativas
Construction *--> ConstructionEmployee : 0..n
Employee *--> ConstructionEmployee : 0..n

Construction *--> ConstructionEquipment : 0..n
Equipment *--> ConstructionEquipment : 0..n

' Inspeção, Relatório e Cidadão
Inspection *--> AuditReport : 1..1
Employee *--> Inspection : 0..n
Citizen *--> Incident : 1..n

' Heranças
Employee -up-|> Person 
Citizen -up-|> Person 

@enduml
